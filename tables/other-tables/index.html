    Other Tables Migration

    Overview

    This document covers the migration of additional tables that provide specialized functionality for campaign management, volunteer tracking, recognition programs, and donor analytics. These tables maintain their core structure while receiving enhancements for better data integrity and reporting capabilities.

    Data Type Enums

    The PostgreSQL implementation uses standard types rather than complex enums for most fields, maintaining simplicity and flexibility:

    Core Data Types Used

    • UUID: All primary keys and foreign key references
    • BOOLEAN: Simple true/false flags (loyal, leader, lots_of_money, etc.)
    • SMALLINT: Numeric scores and ratings (-1 to indicate unset/default)
    • INTEGER: Counts, years, and larger numeric values
    • NUMERIC(18,2): Monetary amounts and precise decimal values
    • VARCHAR(n): Text fields with specific length limits
    • TEXT: Longer text content and comments
    • DATE: Date values without time components
    • TIMESTAMP: Full date and time with timezone support

    MySQL to PostgreSQL Type Mappings

    • tinyint(1) → BOOLEAN (0/1 values become true/false)
    • tinyint(3) → SMALLINT (small numeric values)
    • int(10) → INTEGER or UUID (depending on usage)
    • double → NUMERIC(18,2) (precise decimal arithmetic)
    • varchar(n) → VARCHAR(n) (text with length limits)
    • text → TEXT (unlimited text)
    • date → DATE (date only)
    • timestamp → TIMESTAMP (date and time)
    • year(4) → INTEGER (year values as integers)

    Common Queries

    Prospect Analysis by Wealth Indicators

    -- Analyze prospects by wealth scoring and capacity
SELECT
    p.first_name,
    p.last_name,
    pr.loyal,
    pr.leader,
    pr.lots_of_money,
    pr.wealth_score,
    pr.capacity_to_give,
    pr.networth,
    pr.income,
    pr.ask,
    pr.range_start,
    pr.range_end
FROM prospect pr
JOIN person p ON pr.person_id = p.id
WHERE pr.wealth_score > 5
   OR pr.lots_of_money = true
   OR pr.networth > 1000000
ORDER BY pr.wealth_score DESC, pr.networth DESC;

    Volunteer Training Status Report

    -- Track volunteer training completion across campaigns
SELECT
    p.first_name,
    p.last_name,
    v.volunteer_date,
    v.study_training,
    v.cc_training,
    v.sol_pkg_sent,
    v.chairman,
    v.rank,
    CASE
        WHEN v.study_training IS NOT NULL AND v.cc_training IS NOT NULL
        THEN 'Fully Trained'
        WHEN v.study_training IS NOT NULL
        THEN 'Study Training Only'

    Recognition Summary Report

    -- Comprehensive recognition status across all categories
SELECT
    p.first_name,
    p.last_name,
    hof.is_nominee as hof_nominee,
    hof.is_inductee as hof_inductee,
    hof.inductee_date as hof_date,
    moy.is_recipient as man_of_year,
    moy.award_date as moy_date,
    l.position_type as leadership_position,
    l.start_year as leadership_start,
    l.end_year as leadership_end,
    ds.is_ug_chapter_donor,
    ds.is_school_donor,
    ds.is_hq_donor,
    kp.title as key_player_title
FROM person p
LEFT JOIN hall_of_fame hof ON p.id = hof.person_id
LEFT JOIN man_of_year moy ON p.id = moy.person_id
LEFT JOIN leadership l ON p.id = l.person_id
LEFT JOIN donor_status ds ON p.id = ds.person_id
LEFT JOIN key_players kp ON p.id = kp.person_id
WHERE hof.person_id IS NOT NULL
   OR moy.person_id IS NOT NULL
   OR l.person_id IS NOT NULL
   OR kp.person_id IS NOT NULL
ORDER BY p.last_name, p.first_name;

    Migration Logic and Data Transformation

    Recognition Table Normalization Strategy

    The most complex aspect of this migration is handling the Detail_Frat_Info table from MySQL, which stores all recognition data in a single denormalized table. The PostgreSQL target system normalizes this into separate tables for better data integrity.

    MySQL Source Structure

    -- Single table with boolean flags for different recognition types
CREATE TABLE Detail_Frat_Info (
    donor_id INT,
    hof_hoh_nominee BOOLEAN,
    hof_hoh_nominee_comment VARCHAR(45),
    hof_hoh_inductee BOOLEAN,
    hof_hoh_inductee_comment VARCHAR(45),
    man_of_year BOOLEAN,
    man_of_year_comment VARCHAR(45),
    officer BOOLEAN,
    officer_comment VARCHAR(45),
    board_member BOOLEAN,
    board_member_comment VARCHAR(45),
    -- ... other fields
);

    PostgreSQL Target Structure

    -- Separate normalized tables
CREATE TABLE hall_of_fame (
    person_id UUID PRIMARY KEY,
    is_nominee BOOLEAN,
    nominee_comment VARCHAR(45),
    -- ... other hof fields
);

CREATE TABLE man_of_year (
    person_id UUID PRIMARY KEY,
    is_recipient BOOLEAN,
    comment VARCHAR(45),
    -- ... other moy fields
);

CREATE TABLE leadership (
    id UUID PRIMARY KEY,
    person_id UUID,
    position_type VARCHAR(20) CHECK (position_type IN ('OFFICER', 'BOARD_MEMBER')),
    -- ... other leadership fields
);

    Migration Transformation Logic

    // Pseudo-code for recognition data migration
func migrateRecognitionData(mysqlRecord DetailFratInfo) error {
    personID := convertToUUID(mysqlRecord.DonorID)

    // Migrate Hall of Fame data if any HOF fields are set
    if mysqlRecord.HofHohNominee || mysqlRecord.HofHohInductee {
        hofRecord := HallOfFame{
            PersonID: personID,
            IsNominee: mysqlRecord.HofHohNominee,
            NomineeComment: mysqlRecord.HofHohNomineeComment,
            IsInductee: mysqlRecord.HofHohInductee,
            InducteeComment: mysqlRecord.HofHohInducteeComment,
            // ... other fields
        }
        insertHallOfFameRecord(hofRecord)
    }

    // Migrate Man of Year data if set
    if mysqlRecord.ManOfYear {
        moyRecord := ManOfYear{
            PersonID: personID,
            IsRecipient: mysqlRecord.ManOfYear,
            Comment: mysqlRecord.ManOfYearComment,
            // ... other fields
        }
        insertManOfYearRecord(moyRecord)
    }

    // Migrate leadership data - create separate records for officer and board member
    if mysqlRecord.Officer {
        leadershipRecord := Leadership{
            ID: generateUUID(),
            PersonID: personID,
            PositionType: "OFFICER",
            Comment: mysqlRecord.OfficerComment,
            StartYear: mysqlRecord.OfficerStart,
            EndYear: mysqlRecord.OfficerEnd,
        }
        insertLeadershipRecord(leadershipRecord)
    }

    if mysqlRecord.BoardMember {
        leadershipRecord := Leadership{
            ID: generateUUID(),
            PersonID: personID,
            PositionType: "BOARD_MEMBER",
            Comment: mysqlRecord.BoardMemberComment,
            StartYear: mysqlRecord.BoardMemberStart,
            EndYear: mysqlRecord.BoardMemberEnd,
        }
        insertLeadershipRecord(leadershipRecord)
    }

    return nil
}

    Prospect Data Migration Strategy

    Wealth Scoring System Migration

    -- MySQL prospect data with various scoring systems
-- Map MySQL tinyint(-1) defaults to PostgreSQL SMALLINT with proper null handling
UPDATE prospect SET
    study_action = CASE WHEN study_action = -1 THEN NULL ELSE study_action END,
    study_status = CASE WHEN study_status = -1 THEN NULL ELSE study_status END,
    willingness = CASE WHEN willingness = -1 THEN NULL ELSE willingness END,
    donor_score = CASE WHEN donor_score = -1 THEN NULL ELSE donor_score END,
    ds_rating = CASE WHEN ds_rating = -1 THEN NULL ELSE ds_rating END;

    Campaign Context Migration

    -- Link prospects to campaigns via chapter relationships
-- MySQL stores campaign_id directly, PostgreSQL infers from chapter context
WITH campaign_mapping AS (
    SELECT
        p.id as prospect_id,
        p.person_id,
        o.current_campaign_id
    FROM prospect p
    JOIN person pe ON p.person_id = pe.id
    JOIN person_chapter po ON pe.id = po.person_id
    JOIN chapter o ON po.chapter_id = o.id
    WHERE o.current_campaign_id IS NOT NULL
)
UPDATE prospect p
SET campaign_context = cm.current_campaign_id
FROM campaign_mapping cm
WHERE p.id = cm.prospect_id;

    Volunteer Data Migration Strategy

    Training Date Migration

    -- Migrate training dates with proper null handling
-- MySQL uses '0000-00-00' for unset dates, PostgreSQL uses NULL
UPDATE volunteer SET
    study_training = CASE
        WHEN study_training = '0000-00-00' THEN NULL
        ELSE study_training
    END,
    cc_training = CASE
        WHEN cc_training = '0000-00-00' THEN NULL
        ELSE cc_training
    END,
    sol_pkg_sent = CASE
        WHEN sol_pkg_sent = '0000-00-00' THEN NULL
        ELSE sol_pkg_sent
    END;

    Data Validation and Integrity Checks

    Pre-Migration Validation

    -- Validate person references exist before migration
SELECT
    'prospect' as source_table,
    COUNT(*) as total_records,
    COUNT(CASE WHEN donor_id NOT IN (SELECT id FROM person) THEN 1 END) as invalid_refs
FROM mysql_prospect
UNION ALL
SELECT
    'volunteer',
    COUNT(*),
    COUNT(CASE WHEN donor_id NOT IN (SELECT id FROM person) THEN 1 END)
FROM mysql_volunteer
UNION ALL
SELECT
    'detail_frat_info',
    COUNT(*),
    COUNT(CASE WHEN donor_id NOT IN (SELECT id FROM person) THEN 1 END)
FROM mysql_detail_frat_info;

    Post-Migration Validation

    -- Verify migration completeness
WITH migration_stats AS (
    SELECT
        'prospect' as table_name,
        (SELECT COUNT(*) FROM mysql_prospect) as source_count,
        (SELECT COUNT(*) FROM prospect) as target_count
    UNION ALL
    SELECT
        'volunteer',
        (SELECT COUNT(*) FROM mysql_volunteer),
        (SELECT COUNT(*) FROM volunteer)
    UNION ALL
    SELECT
        'hall_of_fame',
        (SELECT COUNT(*) FROM mysql_detail_frat_info WHERE hof_hoh_nominee = 1 OR hof_hoh_inductee = 1),
        (SELECT COUNT(*) FROM hall_of_fame)
    UNION ALL
    SELECT
        'man_of_year',
        (SELECT COUNT(*) FROM mysql_detail_frat_info WHERE man_of_year = 1),
        (SELECT COUNT(*) FROM man_of_year)
)
SELECT
    table_name,
    source_count,
    target_count,
    CASE
        WHEN source_count = target_count THEN 'PASS'
        ELSE 'FAIL'
    END as migration_status,
    ABS(source_count - target_count) as record_difference
FROM migration_stats;

## Migration Best Practices

### Data Quality Improvements

1. **Relationship Validation**
   ```sql
   -- Ensure all person references exist
   SELECT table_name, COUNT(*) as orphaned_records
   FROM (
       SELECT 'prospect' as table_name FROM prospect p
       LEFT JOIN person pe ON p.person_id = pe.id WHERE pe.id IS NULL
       UNION ALL
       SELECT 'volunteer' FROM volunteer v
       LEFT JOIN person pe ON v.person_id = pe.id WHERE pe.id IS NULL
       -- Add similar checks for other tables
   ) validation
   GROUP BY table_name;
   ```

2. **Date Consistency Checks**
   ```sql
   -- Check for future dates in historical records
   SELECT 'prospect' as table_name, COUNT(*) as future_dates
   FROM prospect WHERE last_contact_date > CURRENT_DATE
   UNION ALL
   SELECT 'volunteer', COUNT(*)
   FROM volunteer WHERE start_date > CURRENT_DATE;
   ```

3. **Enum Value Validation**
   ```sql
   -- Verify enum values are properly mapped
   SELECT 'prospect_rating', prospect_rating, COUNT(*)
   FROM prospect GROUP BY prospect_rating
   UNION ALL
   SELECT 'volunteer_type', volunteer_type::text, COUNT(*)
   FROM volunteer GROUP BY volunteer_type;
   ```

### Performance Optimization

1. **Indexes for Common Queries**
   ```sql
   -- Campaign prospect analysis indexes
   CREATE INDEX idx_prospect_campaign_status ON prospect(campaign_id, prospect_status);
   CREATE INDEX idx_prospect_likelihood ON prospect(likelihood_score);

   -- Volunteer tracking indexes
   CREATE INDEX idx_volunteer_active ON volunteer(is_active);
   CREATE INDEX idx_volunteer_hours ON volunteer(total_hours);

   -- Recognition indexes
   CREATE INDEX idx_hall_of_fame_year ON hall_of_fame(induction_year);
   CREATE INDEX idx_man_of_year_year ON man_of_year(award_year);

   -- Key player analysis indexes
   CREATE INDEX idx_key_players_active ON key_players(is_active);
   CREATE INDEX idx_key_players_influence ON key_players(influence_level, engagement_score);
   ```

2. **Performance Views**
   ```sql
   -- Active volunteer summary
   CREATE VIEW active_volunteers AS
   SELECT
       p.id,
       p.first_name,
       p.last_name,
       v.volunteer_type,
       v.total_hours,
       v.recognition_level
   FROM volunteer v
   JOIN person p ON v.person_id = p.id
   WHERE v.is_active = true;

   -- Campaign prospect summary
   CREATE VIEW campaign_prospects AS
   SELECT
       c.id as campaign_id,
       c.campaign_name,
       COUNT(*) as prospect_count,
       SUM(p.estimated_amount) as total_estimated,
       AVG(p.likelihood_score) as avg_likelihood
   FROM prospect p
   JOIN campaign c ON p.campaign_id = c.id
   GROUP BY c.id, c.campaign_name;
   ```

## Troubleshooting Guide

### Common Migration Issues

#### 1. Recognition Data Normalization Errors

**Issue**: Duplicate key violations when migrating `Detail_Frat_Info` table
    ERROR: duplicate key value violates unique constraint "hall_of_fame_pkey"

    2. Prospect Wealth Score Migration Issues

    Issue: Invalid wealth score values causing check constraint violations 

    ERROR: new row for relation "prospect" violates check constraint "valid_wealth_score"

    Cause: MySQL allows out-of-range values that PostgreSQL constraints reject

    Solution: 

    -- Clean invalid scores before migration
UPDATE mysql_prospect SET
    wealth_score = CASE
        WHEN wealth_score < 0 OR wealth_score > 10 THEN NULL
        ELSE wealth_score
    END,
    donor_score = CASE
        WHEN donor_score < -1 OR donor_score > 10 THEN -1
        ELSE donor_score
    END,
    capacity_to_give = CASE
        WHEN capacity_to_give < 0 OR capacity_to_give > 10 THEN NULL
        ELSE capacity_to_give
    END;

    4. Foreign Key Reference Failures

    Issue: Orphaned records causing foreign key constraint violations 

    ERROR: insert or update on table "prospect" violates foreign key constraint "prospect_person_id_fkey"

    Cause: References to non-existent person records

    Solution: 

    -- Identify and resolve orphaned records
WITH orphaned_prospects AS (
    SELECT p.*
    FROM mysql_prospect p
    LEFT JOIN person pe ON p.donor_id = pe.mysql_id
    WHERE pe.id IS NULL
)
-- Option 1: Skip orphaned records
DELETE FROM mysql_prospect WHERE id IN (SELECT id FROM orphaned_prospects);

-- Option 2: Create placeholder person records
INSERT INTO person (id, first_name, last_name, mysql_id)
SELECT
    gen_random_uuid(),
    'Unknown',
    'Person_' || donor_id,
    donor_id
FROM orphaned_prospects;

    5. Leadership Position Duplication

    Issue: Multiple leadership records for same person and position type 

    ERROR: duplicate key value violates unique constraint "leadership_person_position_unique"

    Cause: Overlapping officer and board member terms in MySQL

    Solution: 

    -- Merge overlapping leadership terms
WITH leadership_consolidated AS (
    SELECT
        donor_id,
        'OFFICER' as position_type,
        MIN(officer_start) as start_year,
        MAX(officer_end) as end_year,
        STRING_AGG(DISTINCT officer_comment, '; ') as comments
    FROM mysql_detail_frat_info
    WHERE officer = 1
    GROUP BY donor_id

    UNION ALL

    SELECT
        donor_id,
        'BOARD_MEMBER',
        MIN(board_member_start),
        MAX(board_member_end),
        STRING_AGG(DISTINCT board_member_comment, '; ')
    FROM mysql_detail_frat_info
    WHERE board_member = 1
    GROUP BY donor_id
)

    Pre-Migration Health Check

    -- Comprehensive pre-migration validation
WITH validation_results AS (
    -- Check for invalid person references
    SELECT
        'Invalid person references in prospect' as check_name,
        COUNT(*) as issue_count,
        'HIGH' as severity
    FROM mysql_prospect p
    LEFT JOIN person pe ON p.donor_id = pe.mysql_id
    WHERE pe.id IS NULL

    UNION ALL

    -- Check for zero dates
    SELECT
        'Zero dates in volunteer table',
        COUNT(*),
        'MEDIUM'
    FROM mysql_volunteer
    WHERE study_training = '0000-00-00'
       OR cc_training = '0000-00-00'
       OR sol_pkg_sent = '0000-00-00'

    UNION ALL

    -- Check for out-of-range scores
    SELECT
        'Invalid wealth scores in prospect',
        COUNT(*),
        'MEDIUM'
    FROM mysql_prospect
    WHERE wealth_score < 0 OR wealth_score > 10
       OR donor_score < -1 OR donor_score > 10

    UNION ALL

    -- Check for duplicate recognition records
    SELECT
        'Duplicate recognition records',
        COUNT(*) - COUNT(DISTINCT donor_id),
        'LOW'
    FROM mysql_detail_frat_info
)
SELECT
    check_name,
    issue_count,
    severity,
    CASE
        WHEN issue_count = 0 THEN 'PASS'
        WHEN severity = 'HIGH' AND issue_count > 0 THEN 'CRITICAL - FIX REQUIRED'
        WHEN severity = 'MEDIUM' AND issue_count > 0 THEN 'WARNING - REVIEW NEEDED'
        ELSE 'INFO - MINOR ISSUES'
    END as status
FROM validation_results
ORDER BY
    CASE severity WHEN 'HIGH' THEN 1 WHEN 'MEDIUM' THEN 2 ELSE 3 END,
    issue_count DESC;

    Post-Migration Verification

    -- Verify migration completeness and accuracy
WITH migration_summary AS (
    SELECT
        'prospect' as table_name,
        (SELECT COUNT(*) FROM mysql_prospect) as source_records,
        (SELECT COUNT(*) FROM prospect) as migrated_records,
        (SELECT COUNT(*) FROM prospect WHERE wealth_score IS NOT NULL) as wealth_scored_records

    UNION ALL

    SELECT
        'volunteer',
        (SELECT COUNT(*) FROM mysql_volunteer),
        (SELECT COUNT(*) FROM volunteer),
        (SELECT COUNT(*) FROM volunteer WHERE study_training IS NOT NULL OR cc_training IS NOT NULL) as trained_volunteers

    UNION ALL

    SELECT
        'hall_of_fame',
        (SELECT COUNT(*) FROM mysql_detail_frat_info WHERE hof_hoh_nominee = 1 OR hof_hoh_inductee = 1),
        (SELECT COUNT(*) FROM hall_of_fame),
        (SELECT COUNT(*) FROM hall_of_fame WHERE is_inductee = true) as inductees

    UNION ALL

    SELECT
        'leadership',
        (SELECT COUNT(*) FROM mysql_detail_frat_info WHERE officer = 1) +
        (SELECT COUNT(*) FROM mysql_detail_frat_info WHERE board_member = 1),
        (SELECT COUNT(*) FROM leadership),
        (SELECT COUNT(DISTINCT person_id) FROM leadership) as unique_leaders
)
SELECT
    table_name,
    source_records,
    migrated_records,
    ROUND((migrated_records::NUMERIC / source_records * 100), 2) as migration_percentage,
    migrated_records - source_records as record_difference,
    CASE
        WHEN migrated_records = source_records THEN 'PERFECT'
        WHEN migrated_records > source_records * 0.95 THEN 'GOOD'
        WHEN migrated_records > source_records * 0.90 THEN 'ACCEPTABLE'
        ELSE 'NEEDS_REVIEW'
    END as migration_quality
FROM migration_summary
ORDER BY migration_percentage DESC;
