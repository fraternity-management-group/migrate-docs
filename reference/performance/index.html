    Performance Tuning Guide

    Overview

    This guide provides comprehensive strategies for optimizing the performance of both the migration process and the resulting PostgreSQL database. Performance tuning is crucial for large datasets and ensuring optimal query response times in production.

    Migration Performance Optimization

    Pre-Migration Preparation

    Migration Process Optimization

    Batch Processing Strategy

    1. Chunked Migration Implementation 

      // Optimal batch size determination
func determineBatchSize(tableSize int64) int {
    switch {
    case tableSize < 10000:
        return 1000
    case tableSize < 100000:
        return 5000
    case tableSize < 1000000:
        return 10000
    default:
        return 25000
    }
}

// Parallel processing with worker pools
func migrateTableInBatches(tableName string, totalRecords int64) error {
    batchSize := determineBatchSize(totalRecords)
    numWorkers := runtime.NumCPU()

    jobs := make(chan BatchJob, numWorkers*2)
    results := make(chan BatchResult, numWorkers*2)

    // Start workers
    for i := 0; i < numWorkers; i++ {
        go batchWorker(jobs, results)
    }

    // Generate batches
    go func() {
        for offset := int64(0); offset < totalRecords; offset += int64(batchSize) {
            jobs <- BatchJob{
                TableName: tableName,
                Offset:    offset,
                Limit:     batchSize,
            }
        }
        close(jobs)
    }()

    // Collect results
    for i := 0; i < int(totalRecords/int64(batchSize))+1; i++ {
        result := <-results
        if result.Error != nil {
            return result.Error
        }
    }

    return nil
}

    2. Memory Management 

      // Efficient memory usage during migration
func processBatchWithMemoryManagement(batch []Record) error {
    const maxMemoryUsage = 100 * 1024 * 1024 // 100MB

    for i, record := range batch {
        if i%1000 == 0 {
            // Force garbage collection periodically
            runtime.GC()

            // Check memory usage
            var m runtime.MemStats
            runtime.ReadMemStats(&m)
            if m.Alloc > maxMemoryUsage {
                return fmt.Errorf("memory usage too high: %d bytes", m.Alloc)
            }
        }

        if err := processRecord(record); err != nil {
            return err
        }
    }
    return nil
}

    Query Optimization

    Common Query Patterns and Optimizations

    1. Donor Search Optimization 

      -- Optimized donor search query
CREATE OR REPLACE FUNCTION search_donors(
    search_term TEXT,
    limit_count INTEGER DEFAULT 50
) RETURNS TABLE(
    person_id UUID,
    full_name TEXT,
    email TEXT,
    total_giving NUMERIC,
    last_gift_date DATE
) AS $$
BEGIN
    RETURN QUERY
    WITH donor_search AS (
        SELECT 
            p.id,
            p.first_name || ' ' || p.last_name as name,
            c.email_address,
            ts_rank(to_tsvector('english', p.first_name || ' ' || p.last_name), 
                   plainto_tsquery('english', search_term)) as rank
        FROM person p
        LEFT JOIN contact c ON p.id = c.person_id AND c.contact_type = 'email' AND c.is_primary = true
        WHERE to_tsvector('english', p.first_name || ' ' || p.last_name) @@ plainto_tsquery('english', search_term)
           OR LOWER(p.email) LIKE LOWER('%' || search_term || '%')
        ORDER BY rank DESC
        LIMIT limit_count
    ),
    giving_summary AS (
        SELECT 
            f.person_id,
            SUM(f.amount) as total_amount,
            MAX(f.date_received) as last_date
        FROM finance f
        WHERE f.person_id IN (SELECT id FROM donor_search)
        GROUP BY f.person_id
    )
    SELECT 
        ds.id,
        ds.name,
        ds.email_address,
        COALESCE(gs.total_amount, 0),
        gs.last_date
    FROM donor_search ds
    LEFT JOIN giving_summary gs ON ds.id = gs.person_id
    ORDER BY ds.rank DESC, gs.total_amount DESC NULLS LAST;
END;
$$ LANGUAGE plpgsql STABLE;

    2. Financial Reporting Optimization 

      -- Materialized view for financial summaries
CREATE MATERIALIZED VIEW donor_giving_summary AS
SELECT 
    p.id as person_id,
    p.first_name,
    p.last_name,
    p.email,
    COUNT(f.id) as gift_count,
    SUM(f.amount) as total_giving,
    AVG(f.amount) as average_gift,
    MIN(f.date_received) as first_gift_date,
    MAX(f.date_received) as last_gift_date,
    SUM(CASE WHEN f.date_received >= DATE_TRUNC('year', CURRENT_DATE) THEN f.amount ELSE 0 END) as current_year_giving,
    SUM(CASE WHEN f.date_received >= DATE_TRUNC('year', CURRENT_DATE) - INTERVAL '1 year' 
              AND f.date_received < DATE_TRUNC('year', CURRENT_DATE) THEN f.amount ELSE 0 END) as prior_year_giving
FROM person p
LEFT JOIN finance f ON p.id = f.person_id
GROUP BY p.id, p.first_name, p.last_name, p.email;

-- Index the materialized view
CREATE UNIQUE INDEX idx_donor_giving_summary_person ON donor_giving_summary(person_id);
CREATE INDEX idx_donor_giving_summary_total ON donor_giving_summary(total_giving DESC);
CREATE INDEX idx_donor_giving_summary_current ON donor_giving_summary(current_year_giving DESC);

-- Refresh procedure
CREATE OR REPLACE FUNCTION refresh_donor_giving_summary() RETURNS VOID AS $$
BEGIN
    REFRESH MATERIALIZED VIEW CONCURRENTLY donor_giving_summary;
END;
$$ LANGUAGE plpgsql;

    3. Contact History Optimization 

      -- Optimized contact log queries with proper indexing
CREATE INDEX CONCURRENTLY idx_contact_log_person_date ON contact_log(person_id, contact_date DESC);
CREATE INDEX CONCURRENTLY idx_contact_log_follow_up ON contact_log(follow_up_date)

-- Function for recent contact activity
CREATE OR REPLACE FUNCTION get_recent_contacts(
    person_uuid UUID,
    days_back INTEGER DEFAULT 90
) RETURNS TABLE(
    contact_date DATE,
    contact_type contact_type_enum,
    staff_member TEXT,
    summary TEXT,
    follow_up_required BOOLEAN
) AS $$
BEGIN
    RETURN QUERY
    SELECT 
        cl.contact_date,
        cl.contact_type,
        s.first_name || ' ' || s.last_name as staff_name,
        cl.summary,
        (cl.follow_up_date IS NOT NULL AND NOT cl.completed) as needs_follow_up
    FROM contact_log cl
    LEFT JOIN person s ON cl.staff_id = s.id
    WHERE cl.person_id = person_uuid
      AND cl.contact_date >= CURRENT_DATE - INTERVAL '%s days' % days_back
    ORDER BY cl.contact_date DESC;
END;
$$ LANGUAGE plpgsql STABLE;

    Connection Pool Optimization

    Application-Level Optimization

    1. Connection Pool Configuration 

      // Optimal connection pool settings
func configureConnectionPool(db *sql.DB) {
    // Set maximum number of open connections
    db.SetMaxOpenConns(25)

    // Set maximum number of idle connections
    db.SetMaxIdleConns(10)

    // Set connection lifetime
    db.SetConnMaxLifetime(5 * time.Minute)

    // Set connection idle time
    db.SetConnMaxIdleTime(30 * time.Second)
}

// Connection health check
func healthCheck(db *sql.DB) error {
    ctx, cancel := context.WithTimeout(context.Background(), 5*time.Second)
    defer cancel()

    if err := db.PingContext(ctx); err != nil {
        return fmt.Errorf("database health check failed: %w", err)
    }

    return nil
}

    2. Query Timeout Management 

      // Query with timeout
func executeQueryWithTimeout(db *sql.DB, query string, timeout time.Duration) (*sql.Rows, error) {
    ctx, cancel := context.WithTimeout(context.Background(), timeout)
    defer cancel()

    stmt, err := db.PrepareContext(ctx, query)
    if err != nil {
        return nil, err
    }
    defer stmt.Close()

    return stmt.QueryContext(ctx)
}

    Monitoring and Maintenance

    Performance Monitoring Queries

    1. Index Usage Analysis 

      -- Check index usage statistics
SELECT 
    schemaname,
    tablename,
    indexname,
    idx_tup_read,
    idx_tup_fetch,
    idx_scan,
    CASE WHEN idx_scan = 0 THEN 'Unused'
         WHEN idx_scan < 10 THEN 'Low Usage'
         ELSE 'Active'
    END as usage_status
FROM pg_stat_user_indexes
ORDER BY idx_scan DESC;

-- Find missing indexes
SELECT 
    schemaname,
    tablename,
    seq_scan,
    seq_tup_read,
    seq_tup_read / seq_scan as avg_tup_per_scan
FROM pg_stat_user_tables
WHERE seq_scan > 0
ORDER BY seq_tup_read DESC;

    2. Query Performance Analysis 

      -- Slow query identification (requires pg_stat_statements)
SELECT 
    query,
    calls,
    total_time,
    mean_time,
    rows,
    100.0 * shared_blks_hit / nullif(shared_blks_hit + shared_blks_read, 0) AS hit_percent
FROM pg_stat_statements
WHERE mean_time > 1000  -- Queries taking more than 1 second on average
ORDER BY total_time DESC
LIMIT 20;

    3. Database Size and Growth Monitoring 

      -- Table size analysis
SELECT 
    schemaname,
    tablename,
    pg_size_pretty(pg_total_relation_size(schemaname||'.'||tablename)) as total_size,
    pg_size_pretty(pg_relation_size(schemaname||'.'||tablename)) as table_size,
    pg_size_pretty(pg_total_relation_size(schemaname||'.'||tablename) - pg_relation_size(schemaname||'.'||tablename)) as index_size
FROM pg_tables
WHERE schemaname = 'public'
ORDER BY pg_total_relation_size(schemaname||'.'||tablename) DESC;

-- Growth tracking
CREATE TABLE IF NOT EXISTS database_size_history (
    recorded_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    database_name TEXT,
    total_size BIGINT,
    table_count INTEGER,
    index_count INTEGER
);

-- Daily size tracking function
CREATE OR REPLACE FUNCTION record_database_size() RETURNS VOID AS $$
BEGIN
    INSERT INTO database_size_history (database_name, total_size, table_count, index_count)
    SELECT 
        current_database(),
        pg_database_size(current_database()),
        (SELECT COUNT(*) FROM pg_tables WHERE schemaname = 'public'),
        (SELECT COUNT(*) FROM pg_indexes WHERE schemaname = 'public');
END;
$$ LANGUAGE plpgsql;

    Automated Maintenance

    1. Vacuum and Analyze Scheduling 

      -- Custom vacuum strategy
CREATE OR REPLACE FUNCTION smart_vacuum_analyze() RETURNS VOID AS $$
DECLARE
    rec RECORD;
BEGIN
    FOR rec IN 
        SELECT schemaname, tablename, n_tup_ins, n_tup_upd, n_tup_del
        FROM pg_stat_user_tables
        WHERE schemaname = 'public'
    LOOP
        -- Vacuum tables with significant changes
        IF rec.n_tup_ins + rec.n_tup_upd + rec.n_tup_del > 1000 THEN
            EXECUTE 'VACUUM ANALYZE ' || rec.schemaname || '.' || rec.tablename;
            RAISE NOTICE 'Vacuumed and analyzed %.%', rec.schemaname, rec.tablename;
        END IF;
    END LOOP;
END;
$$ LANGUAGE plpgsql;

    2. Index Maintenance 

      -- Reindex bloated indexes
CREATE OR REPLACE FUNCTION reindex_bloated_indexes() RETURNS VOID AS $$
DECLARE
    rec RECORD;
BEGIN
    FOR rec IN 
        SELECT schemaname, indexname
        FROM pg_stat_user_indexes
        WHERE idx_scan > 0 
        AND pg_relation_size(indexrelid) > 100 * 1024 * 1024  -- > 100MB
    LOOP
        EXECUTE 'REINDEX INDEX CONCURRENTLY ' || rec.schemaname || '.' || rec.indexname;
        RAISE NOTICE 'Reindexed %.%', rec.schemaname, rec.indexname;
    END LOOP;
END;
$$ LANGUAGE plpgsql;

    2. Concurrent Load Testing 

      // Simulate concurrent users
func TestConcurrentLoad(t *testing.T) {
    db := setupTestDB()
    defer db.Close()

    const numGoroutines = 10
    const queriesPerGoroutine = 100

    var wg sync.WaitGroup
    errChan := make(chan error, numGoroutines)

    for i := 0; i < numGoroutines; i++ {
        wg.Add(1)
        go func(workerID int) {
            defer wg.Done()

            for j := 0; j < queriesPerGoroutine; j++ {
                if err := executeRandomQuery(db); err != nil {
                    errChan <- fmt.Errorf("worker %d, query %d: %w", workerID, j, err)
                    return
                }
            }
        }(i)
    }

    wg.Wait()
    close(errChan)

    for err := range errChan {
        t.Error(err)
    }
}
