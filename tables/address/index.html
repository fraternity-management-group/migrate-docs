    Address Table Migration

    The Address table represents the migration of non-primary address data from MySQL's Address table to PostgreSQL's address table, supporting multiple addresses per person with enhanced international address support.

    Overview

    The address system is split during migration - primary addresses go to the contact table, while additional addresses are stored in the dedicated address table. This provides better normalization and supports multiple address types per person.

    Schema Transformation

    MySQL Source Table: Address

    CREATE TABLE `Address` (
  `aid` int(10) unsigned NOT NULL AUTO_INCREMENT,
  `donor_id` int(10) unsigned NOT NULL DEFAULT 0,
  `street1` varchar(255) NOT NULL DEFAULT '',
  `street2` varchar(255) NOT NULL DEFAULT '',
  `city` varchar(45) NOT NULL DEFAULT '',
  `state` enum('','AL','AK','AS','AZ','AR','CA','CO','CT','DE','DC','FL','GA','GU','HI','ID','IL','IN','IA','KS','KY','LA','ME','MD','MA','MI','MN','MS','MO','MT','NE','NV','NH','NJ','NM','NY','NC','ND','OH','OK','OR','PW','PA','PR','RI','SC','SD','SP','TN','TX','UT','VT','VI','VA','WA','WV','WI','WY') DEFAULT NULL,
  `zip` varchar(45) NOT NULL DEFAULT '',
  `entered_by` varchar(45) DEFAULT NULL,
  `date` date DEFAULT NULL,
  `main` tinyint(1) NOT NULL DEFAULT 0,
  `valid` tinyint(1) DEFAULT NULL,
  `comments` text NOT NULL DEFAULT '',
  `last_changed` timestamp NOT NULL DEFAULT current_timestamp() ON UPDATE current_timestamp(),
  `atype` enum('Home','Business','Summer Home','Winter Home','Other') NOT NULL DEFAULT 'Other',
  `country` varchar(45) DEFAULT NULL,
  PRIMARY KEY (`aid`),
  KEY `Address_FK_Index` (`donor_id`),
  CONSTRAINT `0_1653` FOREIGN KEY (`donor_id`) REFERENCES `Donor` (`id`) ON DELETE CASCADE ON UPDATE NO ACTION
) ENGINE=InnoDB AUTO_INCREMENT=389155 DEFAULT CHARSET=latin1 COLLATE=latin1_swedish_ci;

    PostgreSQL Target Table: address

    CREATE TABLE IF NOT EXISTS address (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  person_id UUID REFERENCES person(id) ON DELETE CASCADE,
  organization_id UUID REFERENCES organization(id) ON DELETE CASCADE,

  -- Address components
  organization text,                -- Organization name (if applicable)
  street text[],                -- Street address (can be multiple lines)

  -- Geographic components
  locality VARCHAR(100) NOT NULL, -- City/Town
  administrative_area VARCHAR(100), -- State/Province/Region
  postal_code VARCHAR(20) NOT NULL,    -- ZIP/postal code
  plus4 VARCHAR(10),                   -- ZIP+4 extension
  country VARCHAR(100) DEFAULT 'United States',
  country_code CHAR(2) DEFAULT 'US',   -- ISO country code

  -- Geocoding and validation
  latitude DECIMAL(10, 8),
  longitude DECIMAL(11, 8),
  geocode_accuracy geocode_accuracy_enum,        -- ROOFTOP, RANGE_INTERPOLATED, etc.

  -- Address metadata
  address_type address_type_enum NOT NULL DEFAULT 'Home',
  is_primary BOOLEAN NOT NULL DEFAULT FALSE,
  is_mailing BOOLEAN NOT NULL DEFAULT TRUE,
  validation_status validation_status_enum,       -- VERIFIED, UNVERIFIED, INVALID
  validation_date TIMESTAMP,

  -- Administrative fields
  effective_date DATE DEFAULT CURRENT_DATE,
  end_date DATE,
  entered_by VARCHAR(100),
  created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
  last_changed TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
  comments TEXT,

  -- Constraints
  CONSTRAINT chk_address_owner CHECK (
    (person_id IS NOT NULL AND organization_id IS NULL) OR
    (person_id IS NULL AND organization_id IS NOT NULL)
  ),
  CONSTRAINT chk_date_range CHECK (end_date IS NULL OR effective_date <= end_date),
  CONSTRAINT chk_coordinates CHECK (
    (latitude IS NULL AND longitude IS NULL) OR
    (latitude IS NOT NULL AND longitude IS NOT NULL AND
     latitude BETWEEN -90 AND 90 AND longitude BETWEEN -180 AND 180)
  )
);

-- Create partial unique indexes for primary addresses (only applies when is_primary = true)
CREATE UNIQUE INDEX idx_primary_address_person ON address (person_id) WHERE is_primary = true;
CREATE UNIQUE INDEX idx_primary_address_org ON address (organization_id) WHERE is_primary = true;

    PostgreSQL Address Type Enum

    CREATE TYPE address_type_enum AS ENUM (
  'Home', 'Business', 'Mailing', 'Billing', 'Shipping',
  'Summer Home', 'Winter Home', 'Previous', 'Other'

    Key Changes and Improvements

    1. Primary Key Modernization

    MySQL: Auto-increment integer 

    aid INT AUTO_INCREMENT PRIMARY KEY

    PostgreSQL: UUID for better distributed system support 

    id UUID PRIMARY KEY DEFAULT gen_random_uuid()

    2. Enhanced Data Structure

    Address Components Enhancement

    MySQL: Traditional street fields 

    street1 varchar(255),
street2 varchar(255),
city varchar(45)

    PostgreSQL: Modern address structure with international support 

    organization text,                     -- Organization name (if applicable)
street text[],                        -- Street address (can be multiple lines)
locality VARCHAR(100) NOT NULL,       -- City/Town
administrative_area VARCHAR(100)      -- State/Province/Region

    International Address Support

    MySQL: Limited postal code and country support 

    state ENUM(...),

    PostgreSQL: Comprehensive international support 

    postal_code VARCHAR(20) NOT NULL,     -- ZIP/postal code
plus4 VARCHAR(10),                    -- ZIP+4 extension
country VARCHAR(100) DEFAULT 'United States',
country_code CHAR(2) DEFAULT 'US'     -- ISO country code

    Enhanced Address Types

    MySQL: Basic enum 

    atype enum('Home','Business','Summer Home','Winter Home','Other')

    PostgreSQL: Extended custom enum with additional types 

    address_type address_type_enum NOT NULL DEFAULT 'Home'

-- 'Summer Home', 'Winter Home', 'Previous', 'Other'

    Migration Logic

    1. UUID Generation and Mapping

    -- Create mapping for address records
INSERT INTO id_mapping (mysql_table, mysql_id, postgres_id)
SELECT 'address', aid, gen_random_uuid()
FROM mysql.Address
WHERE NOT is_primary_address(donor_id, aid);

    2. Person ID Resolution

    A mapping of Donor.id to Person.id is stored during the migration. This mapping is available in the personMapping field in the checkpoint file.

    3. Data Validation and Cleanup

    During migration the github.com/Boostport/address library is used to validate and normalize addresses from MySQL.

    Common Queries

    Person Address Management

    -- Get all addresses for a person
SELECT id, address_type, street, locality, administrative_area, postal_code, country, is_primary
FROM address
WHERE person_id = $1
  AND (validation_status IS NULL OR validation_status != 'INVALID')
ORDER BY is_primary DESC, address_type;

-- Find primary address for a person
SELECT street, locality, administrative_area, postal_code, country
FROM address
WHERE person_id = $1
  AND is_primary = true
  AND (validation_status IS NULL OR validation_status != 'INVALID');

-- Count addresses per person
SELECT person_id, COUNT(*) as address_count
FROM address
WHERE validation_status != 'INVALID' OR validation_status IS NULL
GROUP BY person_id
HAVING COUNT(*) > 1
ORDER BY address_count DESC;

    Geographic Queries

    -- Find people in a specific city/state
SELECT DISTINCT p.first_name, p.last_name, a.locality, a.administrative_area
FROM person p
JOIN address a ON p.id = a.person_id
WHERE a.locality = 'San Francisco'
  AND a.administrative_area = 'CA'
  AND (a.validation_status IS NULL OR a.validation_status != 'INVALID')
ORDER BY p.last_name, p.first_name;

-- Address distribution by administrative area (state/province)
SELECT administrative_area, COUNT(*) as address_count
FROM address
WHERE (validation_status IS NULL OR validation_status != 'INVALID')
  AND country = 'United States'
GROUP BY administrative_area
ORDER BY address_count DESC;

-- International addresses
SELECT country, COUNT(*) as address_count
FROM address
WHERE (validation_status IS NULL OR validation_status != 'INVALID')
  AND country != 'United States'
GROUP BY country
ORDER BY address_count DESC;

-- Addresses within geographic bounds (using geocoding)
SELECT id, street, locality, administrative_area, latitude, longitude
FROM address
WHERE latitude BETWEEN 37.7 AND 37.8
  AND longitude BETWEEN -122.5 AND -122.4
  AND latitude IS NOT NULL
ORDER BY latitude, longitude;

    Organization Address Queries

    -- Get all addresses for an organization
SELECT id, address_type, street, locality, administrative_area, postal_code
FROM address
WHERE organization_id = $1
  AND (validation_status IS NULL OR validation_status != 'INVALID')
ORDER BY is_primary DESC, address_type;

-- Organizations with multiple locations
SELECT organization_id, COUNT(*) as location_count
FROM address
WHERE organization_id IS NOT NULL
  AND (validation_status IS NULL OR validation_status != 'INVALID')
GROUP BY organization_id
HAVING COUNT(*) > 1
ORDER BY location_count DESC;

    Data Quality Queries

    -- Addresses needing verification
SELECT a.id, p.first_name, p.last_name, a.street, a.locality, a.administrative_area
FROM address a
LEFT JOIN person p ON a.person_id = p.id
WHERE (a.validation_status IS NULL OR a.validation_status = 'UNVERIFIED')
  AND (a.validation_date IS NULL OR a.validation_date < NOW() - INTERVAL '1 year')
ORDER BY a.validation_date NULLS FIRST;

-- Invalid or problematic addresses
SELECT a.id,
       COALESCE(p.first_name || ' ' || p.last_name, o.title) as owner,
       a.street, a.locality, a.comments
FROM address a
LEFT JOIN person p ON a.person_id = p.id
LEFT JOIN organization o ON a.organization_id = o.id
WHERE a.validation_status = 'INVALID'
ORDER BY a.last_changed DESC;

    Data Quality Improvements

    1. Address Validation

    -- Add postal code format constraints
ALTER TABLE address
ADD CONSTRAINT chk_address_postal_code_format
CHECK (
    (country = 'United States' AND postal_code ~ '^\d{5}(-\d{4})?$') OR
    country != 'United States' OR
    postal_code IS NULL
);

-- Ensure required fields
ALTER TABLE address
ADD CONSTRAINT chk_address_required_fields
CHECK (
    street IS NOT NULL AND
    array_length(street, 1) > 0 AND
    locality IS NOT NULL AND
    trim(locality) != ''
);

-- Validate geocoding coordinates (already included in table definition)
-- latitude BETWEEN -90 AND 90 AND longitude BETWEEN -180 AND 180

    2. Primary Address Management

    -- Function to ensure only one primary address per person
CREATE OR REPLACE FUNCTION ensure_single_primary_address()
RETURNS TRIGGER AS $$
BEGIN
    IF NEW.is_primary = true THEN
        -- Set all other addresses for this person to non-primary
        UPDATE address
        SET is_primary = false
        WHERE person_id = NEW.person_id
          AND id != NEW.id;
    END IF;

    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER address_primary_management
    AFTER INSERT OR UPDATE ON address
    FOR EACH ROW
    WHEN (NEW.is_primary = true)
    EXECUTE FUNCTION ensure_single_primary_address();

    3. International Address Support

    // Go functions for international address handling
func normalizePostalCode(code, country string) string {
    switch country {
    case "United States":
        return normalizeUSZipCode(code)
    case "Canada":
        return normalizeCanadianPostalCode(code)
    case "United Kingdom":
        return normalizeUKPostcode(code)
    default:
        return strings.TrimSpace(strings.ToUpper(code))
    }
}

func isValidCountry(country string) bool {
    validCountries := []string{
        "United States", "Canada", "United Kingdom",
        "Australia", "New Zealand", "Germany", "France",
        // ... additional countries
    }

    for _, valid := range validCountries {
        if strings.EqualFold(country, valid) {
            return true
        }
    }
    return false
}

    Troubleshooting

    Common Migration Issues

    1. Missing Person References

    -- Find addresses with missing person references
SELECT a.id, a.street1, a.city, a.state
FROM address a
LEFT JOIN person p ON a.person_id = p.id
WHERE p.id IS NULL;

    2. Multiple Primary Addresses

    -- Check for people with multiple primary addresses
SELECT person_id, COUNT(*) as primary_count
FROM address
WHERE is_primary = true
GROUP BY person_id
HAVING COUNT(*) > 1;

    3. Invalid Address Data

    -- Find addresses with potential data quality issues
SELECT id, street, locality, administrative_area, postal_code, country
FROM address
WHERE (street IS NULL OR array_length(street, 1) = 0)
   OR (locality IS NULL OR trim(locality) = '')
   OR (country = 'United States' AND (administrative_area IS NULL OR length(administrative_area) != 2))
   OR (country = 'United States' AND postal_code !~ '^\d{5}(-\d{4})?$')
   OR (validation_status = 'INVALID');

    Performance Optimization

    Query Optimization

    • Use person_id index for person-based address lookups
    • Use composite indexes for location-based searches
    • Consider partial indexes for frequently queried address types

    Data Management

    • Regular address validation and cleanup processes
    • Batch updates for address verification status
    • Archive old or invalid addresses instead of deleting
