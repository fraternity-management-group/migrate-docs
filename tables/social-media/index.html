    Social Media Table Migration

    The Social Media table represents the extraction and normalization of social media fields from MySQL's Donor table into a dedicated PostgreSQL social_media table, providing better chapter and extensibility for social media platform management.

    Overview

    Social media information that was previously embedded as individual fields in the MySQL Donor table is now properly normalized into a dedicated social_media table. This allows for multiple social media profiles per person and better management of various platforms.

    Schema Transformation

    MySQL Source Fields (from Donor table)

    -- Social media fields embedded in Donor table
twitter VARCHAR(15),
linkedin VARCHAR(45),
facebook VARCHAR(45),
instagram VARCHAR(45)

    PostgreSQL Target Table: social_media

    CREATE TABLE IF NOT EXISTS social_media (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  person_id UUID NOT NULL REFERENCES person(id) ON DELETE CASCADE,
  service social_media_service_enum NOT NULL,
  custom_service VARCHAR(45), -- Used when service = 'Other'
  handle VARCHAR(100), -- Username/handle (without @ symbol)
  display_name VARCHAR(100), -- Display name on the platform
  profile_url TEXT,

  -- Platform engagement data
  is_verified BOOLEAN NOT NULL DEFAULT FALSE,
  is_active BOOLEAN NOT NULL DEFAULT TRUE,

  -- Privacy and contact preferences
  is_public_profile BOOLEAN NOT NULL DEFAULT TRUE,
  allows_messaging BOOLEAN NOT NULL DEFAULT TRUE,
  allows_mentions BOOLEAN NOT NULL DEFAULT TRUE,

  -- Metadata
  platform_user_id VARCHAR(100), -- Platform's internal user ID

  created_at TIMESTAMPTZ NOT NULL DEFAULT CURRENT_TIMESTAMP,
  last_changed TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,

  -- Constraints
  CONSTRAINT chk_custom_service_required CHECK (
    (service != 'Other') OR (service = 'Other' AND custom_service IS NOT NULL)
  ),
  CONSTRAINT chk_url_format CHECK (
    profile_url IS NULL OR profile_url ~ '^https?://'
  )
);

    PostgreSQL Social Media Service Enum

    CREATE TYPE social_media_service_enum AS ENUM (
  'Facebook', 'Twitter', 'LinkedIn', 'Instagram', 'YouTube', 'TikTok',
  'Snapchat', 'Pinterest', 'Reddit', 'Discord', 'Mastodon', 'Threads',
  'WhatsApp', 'Telegram', 'Signal', 'GitHub', 'GitLab', 'Stack Overflow',
  'Medium', 'Substack', 'Twitch', 'Other'
);
    'Snapchat',
    'Pinterest',
    'Discord',
    'Reddit',
    'Other'
);

    Migration Logic

    1. Platform Mapping

    Each MySQL field maps to a specific platform in PostgreSQL:

    // Go migration logic for platform mapping
func migratePersonSocialMedia(donorID int, donor *MySQLDonor) error {
    personID, err := mapPersonID(donorID)
    if err != nil {
        return err
    }

    // Migrate Twitter
    if donor.TwitterHandle != "" {
        err = createSocialMediaRecord(personID, "Twitter", donor.TwitterHandle, "")
        if err != nil {
            return err
        }
    }

    // Migrate LinkedIn
    if donor.LinkedInURL != "" {
        err = createSocialMediaRecord(personID, "LinkedIn", "", donor.LinkedInURL)
        if err != nil {
            return err
        }
    }

    // Migrate Facebook
    if donor.FacebookURL != "" {
        err = createSocialMediaRecord(personID, "Facebook", "", donor.FacebookURL)
        if err != nil {
            return err
        }
    }

    // Migrate Instagram
    if donor.InstagramHandle != "" {
        err = createSocialMediaRecord(personID, "Instagram", donor.InstagramHandle, "")
        if err != nil {
            return err
        }
    }

    return nil
}

    2. Handle vs URL Detection

    The migration intelligently determines whether the data is a handle or URL:

    func createSocialMediaRecord(personID uuid.UUID, platform, data, url string) error {
    socialMedia := &SocialMedia{
        ID:       uuid.New(),
        PersonID: personID,
        Platform: platform,
        IsPublic: true, // Default assumption
    }

    // Determine if data is a URL or handle
    if strings.HasPrefix(data, "http") || strings.Contains(data, ".com") {
        socialMedia.URL = normalizeURL(data)
        socialMedia.Handle = extractHandleFromURL(data, platform)
    } else {
        socialMedia.Handle = normalizeHandle(data, platform)
        socialMedia.URL = generateURLFromHandle(data, platform)
    }

    return insertSocialMedia(socialMedia)
}

    3. URL Normalization

    func normalizeURL(url string) string {
    // Ensure proper protocol
    if !strings.HasPrefix(url, "http") {
        url = "https://" + url
    }

    // Remove trailing slashes
    url = strings.TrimRight(url, "/")

    return url
}

func normalizeHandle(handle, platform string) string {
    // Remove @ symbols
    handle = strings.TrimPrefix(handle, "@")

    // Platform-specific normalization
    switch platform {
    case "Twitter":
        return normalizeTwitterHandle(handle)
    case "Instagram":
        return normalizeInstagramHandle(handle)
    case "LinkedIn":
        return normalizeLinkedInHandle(handle)
    default:
        return strings.ToLower(strings.TrimSpace(handle))
    }
}

    4. URL Generation

    func generateURLFromHandle(handle, platform string) string {
    baseURLs := map[string]string{
        "Twitter":   "https://twitter.com/",
        "Instagram": "https://instagram.com/",
        "LinkedIn":  "https://linkedin.com/in/",
        "Facebook":  "https://facebook.com/",
        "YouTube":   "https://youtube.com/@",
        "TikTok":    "https://tiktok.com/@",
    }

    if baseURL, exists := baseURLs[platform]; exists {
        return baseURL + strings.TrimPrefix(handle, "@")
    }

    return ""
}

    Indexes and Performance

    Primary and Foreign Key Indexes

    -- Primary key index (automatic)
CREATE INDEX CONCURRENTLY IF NOT EXISTS idx_social_media_pkey ON social_media(id);

-- Foreign key index for person lookups
CREATE INDEX CONCURRENTLY IF NOT EXISTS idx_social_media_person_id ON social_media(person_id);

-- Platform-based queries
CREATE INDEX CONCURRENTLY IF NOT EXISTS idx_social_media_platform ON social_media(platform);

-- Handle searches
CREATE INDEX CONCURRENTLY IF NOT EXISTS idx_social_media_handle ON social_media(handle) WHERE handle IS NOT NULL;

    Search and Query Optimization

    -- URL-based searches
CREATE INDEX CONCURRENTLY IF NOT EXISTS idx_social_media_url ON social_media(url) WHERE url IS NOT NULL;

-- Public profile searches
CREATE INDEX CONCURRENTLY IF NOT EXISTS idx_social_media_public ON social_media(platform, is_public) WHERE is_public = true;

-- Full-text search for handles and notes
CREATE INDEX CONCURRENTLY IF NOT EXISTS idx_social_media_search
ON social_media USING GIN(to_tsvector('english', COALESCE(handle, '') || ' ' || COALESCE(notes, '')));

    Common Queries

    Person Social Media Profiles

    -- Get all social media profiles for a person
SELECT sm.platform, sm.handle, sm.url, sm.is_public, sm.is_verified
FROM social_media sm
WHERE sm.person_id = $1
ORDER BY sm.platform;

-- Find people on specific platforms
SELECT p.first_name, p.last_name, sm.handle, sm.url
FROM person p
JOIN social_media sm ON p.id = sm.person_id
WHERE sm.platform = 'Twitter'
  AND sm.is_public = true
ORDER BY p.last_name, p.first_name;

-- Count social media presence by platform
SELECT sm.platform, COUNT(*) as user_count
FROM social_media sm
WHERE sm.is_public = true
GROUP BY sm.platform
ORDER BY user_count DESC;

    Search and Discovery

    -- Search for social media handles
SELECT p.first_name, p.last_name, sm.platform, sm.handle
FROM person p
JOIN social_media sm ON p.id = sm.person_id
WHERE sm.handle ILIKE '%smith%'
  AND sm.is_public = true
ORDER BY sm.platform, sm.handle;

-- Find verified accounts
SELECT p.first_name, p.last_name, sm.platform, sm.handle, sm.follower_count
FROM person p
JOIN social_media sm ON p.id = sm.person_id
WHERE sm.is_verified = true
ORDER BY sm.follower_count DESC NULLS LAST;

-- Recently updated profiles
SELECT p.first_name, p.last_name, sm.platform, sm.last_updated
FROM person p
JOIN social_media sm ON p.id = sm.person_id
WHERE sm.last_updated >= NOW() - INTERVAL '30 days'
ORDER BY sm.last_updated DESC;

    Platform Analytics

    -- Social media adoption by platform
SELECT
    sm.platform,
    COUNT(*) as total_profiles,
    COUNT(CASE WHEN sm.is_public THEN 1 END) as public_profiles,
    COUNT(CASE WHEN sm.is_verified THEN 1 END) as verified_profiles,
    AVG(sm.follower_count) as avg_followers
FROM social_media sm
GROUP BY sm.platform
ORDER BY total_profiles DESC;

-- People with multiple social media platforms
SELECT
    p.first_name,
    p.last_name,
    COUNT(sm.id) as platform_count,
    string_agg(sm.platform::text, ', ' ORDER BY sm.platform) as platforms
FROM person p
JOIN social_media sm ON p.id = sm.person_id
GROUP BY p.id, p.first_name, p.last_name
HAVING COUNT(sm.id) > 1
ORDER BY platform_count DESC, p.last_name;

    Data Quality Improvements

    1. Platform-Specific Validation

    -- Twitter handle validation
ALTER TABLE social_media
ADD CONSTRAINT chk_social_media_twitter_handle
CHECK (
    platform != 'Twitter' OR
    (handle IS NOT NULL AND handle ~ '^[A-Za-z0-9_]{1,15}$')
);

-- LinkedIn URL validation
ALTER TABLE social_media
ADD CONSTRAINT chk_social_media_linkedin_url
CHECK (
    platform != 'LinkedIn' OR
    (url IS NOT NULL AND url ~ '^https?://.*linkedin\.com/')
);

-- Instagram handle validation
ALTER TABLE social_media
ADD CONSTRAINT chk_social_media_instagram_handle
CHECK (
    platform != 'Instagram' OR
    (handle IS NOT NULL AND handle ~ '^[A-Za-z0-9_.]{1,30}$')
);

    2. Data Consistency

    -- Ensure either handle or URL is provided
ALTER TABLE social_media
ADD CONSTRAINT chk_social_media_handle_or_url
CHECK (handle IS NOT NULL OR url IS NOT NULL);

-- Follower count validation
ALTER TABLE social_media
ADD CONSTRAINT chk_social_media_follower_count
CHECK (follower_count IS NULL OR follower_count >= 0);

    3. Update Triggers

    -- Function to sync handle and URL
CREATE OR REPLACE FUNCTION sync_social_media_handle_url()
RETURNS TRIGGER AS $$
BEGIN
    -- Generate URL from handle if URL is missing
    IF NEW.url IS NULL AND NEW.handle IS NOT NULL THEN
        NEW.url = generate_url_from_handle(NEW.handle, NEW.platform);
    END IF;

    -- Extract handle from URL if handle is missing
    IF NEW.handle IS NULL AND NEW.url IS NOT NULL THEN
        NEW.handle = extract_handle_from_url(NEW.url, NEW.platform);
    END IF;

    -- Update timestamp
    NEW.updated_at = NOW();

    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER social_media_sync_handle_url
    BEFORE INSERT OR UPDATE ON social_media
    FOR EACH ROW
    EXECUTE FUNCTION sync_social_media_handle_url();

    Advanced Features

    1. Social Media Analytics

    // Go functions for social media analytics
type SocialMediaStats struct {
    TotalProfiles    int
    PlatformBreakdown map[string]int
    VerifiedCount    int
    AverageFollowers float64
}

func GetSocialMediaStats(db *sql.DB) (*SocialMediaStats, error) {
    stats := &SocialMediaStats{
        PlatformBreakdown: make(map[string]int),
    }

    query := `
        SELECT
            COUNT(*) as total,
            platform,
            COUNT(CASE WHEN is_verified THEN 1 END) as verified,
            AVG(follower_count) as avg_followers
        FROM social_media
        GROUP BY platform
    `

    rows, err := db.Query(query)
    if err != nil {
        return nil, err
    }
    defer rows.Close()

    for rows.Next() {
        var platform string
        var count, verified int
        var avgFollowers sql.NullFloat64

        err := rows.Scan(&count, &platform, &verified, &avgFollowers)
        if err != nil {
            return nil, err
        }

        stats.TotalProfiles += count
        stats.PlatformBreakdown[platform] = count
        stats.VerifiedCount += verified

        if avgFollowers.Valid {
            stats.AverageFollowers = avgFollowers.Float64
        }
    }

    return stats, nil
}

    2. Profile Validation

    func ValidateSocialMediaProfile(platform, handle, url string) error {
    switch platform {
    case "Twitter":
        return validateTwitterProfile(handle, url)
    case "Instagram":
        return validateInstagramProfile(handle, url)
    case "LinkedIn":
        return validateLinkedInProfile(handle, url)
    case "Facebook":
        return validateFacebookProfile(handle, url)
    default:
        return validateGenericProfile(handle, url)
    }
}

func validateTwitterProfile(handle, url string) error {
    if handle != "" && !regexp.MustCompile(`^[A-Za-z0-9_]{1,15}$`).MatchString(handle) {
        return errors.New("invalid Twitter handle format")
    }

    if url != "" && !strings.Contains(url, "twitter.com") {
        return errors.New("invalid Twitter URL")
    }

    return nil
}

    Troubleshooting

    Common Migration Issues

    1. Invalid Social Media Data

    -- Find profiles with invalid handles or URLs
SELECT sm.id, p.first_name, p.last_name, sm.platform, sm.handle, sm.url
FROM social_media sm
JOIN person p ON sm.person_id = p.id
WHERE (sm.handle IS NULL AND sm.url IS NULL)
   OR (sm.platform = 'Twitter' AND sm.handle !~ '^[A-Za-z0-9_]{1,15}$')
   OR (sm.platform = 'LinkedIn' AND sm.url !~ '^https?://.*linkedin\.com/');

    2. Duplicate Platform Entries

    -- Check for duplicate platform entries per person (should not exist due to unique constraint)
SELECT person_id, platform, COUNT(*) as duplicate_count
FROM social_media
GROUP BY person_id, platform
HAVING COUNT(*) > 1;

    3. Missing Person References

    -- Find social media records with missing person references
SELECT sm.id, sm.platform, sm.handle
FROM social_media sm
LEFT JOIN person p ON sm.person_id = p.id
WHERE p.id IS NULL;
